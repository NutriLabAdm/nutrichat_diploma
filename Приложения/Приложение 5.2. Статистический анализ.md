## 5.2.1. Сравнение групп

### t-критерий Стьюдента

```python
# Статистический тест для сравнения групп
from scipy import stats
import numpy as np

# Данные симуляций
group_a_responses = [...]  # 1000 значений отклика группы A (случайные Twin)
group_b_responses = [...]  # 1000 значений отклика группы B (интервью-основанные Twin)

# t-тест для независимых выборок
t_stat, p_value = stats.ttest_ind(group_a_responses, group_b_responses)

# Вычисление разности средних
mean_a = np.mean(group_a_responses)
mean_b = np.mean(group_b_responses)
mean_diff = mean_b - mean_a

# Стандартная ошибка разности
se_a = np.std(group_a_responses, ddof=1) / np.sqrt(len(group_a_responses))
se_b = np.std(group_b_responses, ddof=1) / np.sqrt(len(group_b_responses))
se_diff = np.sqrt(se_a**2 + se_b**2)

# Доверительный интервал 95%
ci_lower = mean_diff - 1.96 * se_diff
ci_upper = mean_diff + 1.96 * se_diff

print(f"t-statistic: {t_stat:.3f}")
print(f"p-value: {p_value:.6f}")
print(f"Mean difference: {mean_diff:.3f}")
print(f"95% CI: [{ci_lower:.3f}, {ci_upper:.3f}]")
```

### Chi-square тест для conversion rate

```python
from scipy.stats import chi2_contingency

# Таблица сопряжённости
# [успехи_контроль, неуспехи_контроль]
# [успехи_эксперимент, неуспехи_эксперимент]
contingency_table = [
    [250, 750],   # Группа A: 250 успехов из 1000
    [320, 680]    # Группа B: 320 успехов из 1000
]

chi2, p_value_chi, dof, expected = chi2_contingency(contingency_table)

print(f"Chi-square: {chi2:.3f}")
print(f"p-value: {p_value_chi:.6f}")
```

### Uplift Modeling (Mann-Whitney U)

```python
from scipy.stats import mannwhitneyu

# Непараметрический тест для оценки uplift
u_stat, p_value_u = mannwhitneyu(
    group_b_responses, 
    group_a_responses, 
    alternative='greater'  # B > A
)

# Расчёт uplift
uplift = (np.mean(group_b_responses) - np.mean(group_a_responses)) / np.mean(group_a_responses)

print(f"U-statistic: {u_stat:.3f}")
print(f"p-value (one-sided): {p_value_u:.6f}")
print(f"Uplift: {uplift:.2%}")
```

## 5.2.2. Расчёт Effect Size

### Cohen's d

```python
def cohens_d(group1, group2):
    """Расчёт Cohen's d для двух групп"""
    n1, n2 = len(group1), len(group2)
    var1, var2 = np.var(group1, ddof=1), np.var(group2, ddof=1)
    
    # Объединённое стандартное отклонение
    pooled_std = np.sqrt(((n1-1)*var1 + (n2-1)*var2) / (n1+n2-2))
    
    # Cohen's d
    d = (np.mean(group2) - np.mean(group1)) / pooled_std
    
    return d

d = cohens_d(group_a_responses, group_b_responses)

# Интерпретация
if abs(d) < 0.2:
    effect = "незначительный"
elif abs(d) < 0.5:
    effect = "малый"
elif abs(d) < 0.8:
    effect = "средний"
else:
    effect = "большой"

print(f"Cohen's d: {d:.3f} ({effect} эффект)")
```

## 5.2.3. Множественные сравнения

### Коррекция Бонферрони

```python
from scipy.stats import false_discovery_control

# Список p-значений для множественных сравнений
p_values = [0.001, 0.015, 0.032, 0.045, 0.120]

# Коррекция Бонферрони
alpha = 0.05
n_tests = len(p_values)
alpha_corrected = alpha / n_tests

significant = [p < alpha_corrected for p in p_values]

print(f"Alpha (скорректированный): {alpha_corrected:.4f}")
print(f"Значимые результаты: {sum(significant)} из {n_tests}")
```

## 5.2.4. Мощность теста

### Расчёт мощности

```python
from scipy.stats import norm

def calculate_power(effect_size, n, alpha=0.05):
    """Расчёт статистической мощности"""
    z_alpha = norm.ppf(1 - alpha/2)
    z_beta = (effect_size * np.sqrt(n/2)) - z_alpha
    power = norm.cdf(z_beta)
    return power

# Расчёт для различных сценариев
for n in [100, 500, 1000]:
    for d in [0.2, 0.3, 0.5]:
        power = calculate_power(d, n)
        print(f"n={n}, d={d}: power={power:.2%}")
```
