# Приложение 5.3. Модель экстраполяции LTV

## Введение

Приложение содержит детальное описание модели экстраполяции результатов эксперимента на LTV реальных пользователей.

---

## 5.3.1. Основная функция расчёта

```python
import numpy as np
from scipy import stats

def calculate_lift_from_twin_experiment(twin_results: dict, real_users_data: dict) -> dict:
    """
    Расчёт влияния интервенций на LTV на основе результатов эксперимента на Twin
    
    Args:
        twin_results: результаты симуляции на Twin {
            'uplift': float,          # uplift от интервенции (например, 0.23)
            'engagement_diff': float, # разница в engagement
            'conversion_diff': float  # разница в конверсии
        }
        real_users_data: исторические данные реальных пользователей {
            'avg_ltv': float,         # средний LTV (например, 5000)
            'avg_engagement': float,  # средний engagement
            'engagement_to_ltv_corr': float  # корреляция engagement-LTV
        }
    
    Returns:
        словарь с проекцией LTV и доверительным интервалом
    """
    
    # 1. Калибровка: сопоставление поведения Twin и реальных пользователей
    # Используем корреляцию между поведением Twin и историческими данными
    calibration_factor = min(real_users_data['engagement_to_ltv_corr'], 0.7)
    
    # 2. Применение uplift от интервенции с калибровкой
    base_ltv = real_users_data['avg_ltv']
    intervention_uplift = twin_results['uplift'] * calibration_factor
    
    projected_ltv = base_ltv * (1 + intervention_uplift)
    
    # 3. Расчёт доверительного интервала
    # Стандартная ошибка проекции
    se = base_ltv * twin_results['uplift'] * np.sqrt(1 - calibration_factor**2)
    
    ci_95 = stats.t.interval(
        0.95, 
        df=len(real_users_data) - 1,
        loc=projected_ltv,
        scale=se
    )
    
    return {
        'projected_ltv': round(projected_ltv, 2),
        'uplift_percent': round(intervention_uplift * 100, 1),
        'ci_95_lower': round(ci_95[0], 2),
        'ci_95_upper': round(ci_95[1], 2),
        'calibration_factor': round(calibration_factor, 2)
    }
```

## 5.3.2. Пример использования

```python
# Данные эксперимента на Twin
twin_results = {
    'uplift': 0.23,           # +23% engagement
    'engagement_diff': 0.15,
    'conversion_diff': 0.08
}

# Исторические данные реальных пользователей
real_users_data = {
    'avg_ltv': 5000,          # 5000 руб средний LTV
    'avg_engagement': 0.35,   # 35% engagement
    'engagement_to_ltv_corr': 0.65  # корреляция r = 0.65
}

# Расчёт проекции LTV
result = calculate_lift_from_twin_experiment(twin_results, real_users_data)

print(f"Базовый LTV: {real_users_data['avg_ltv']} руб")
print(f"Проектируемый LTV: {result['projected_ltv']} руб")
print(f"Uplift: +{result['uplift_percent']}%")
print(f"95% CI: [{result['ci_95_lower']}, {result['ci_95_upper']}]")
```

## 5.3.3. Чувствительность модели

```python
def sensitivity_analysis(twin_uplift_range, real_data):
    """
    Анализ чувствительности проекции LTV
    """
    results = []
    
    for uplift in twin_uplift_range:
        twin_results = {'uplift': uplift, 'engagement_diff': 0, 'conversion_diff': 0}
        result = calculate_lift_from_twin_experiment(twin_results, real_data)
        results.append({
            'twin_uplift': uplift,
            'projected_ltv': result['projected_ltv'],
            'ltv_uplift': result['uplift_percent']
        })
    
    return results

# Диапазон uplift от 0.05 до 0.40
uplift_range = np.arange(0.05, 0.45, 0.05)
sensitivity = sensitivity_analysis(uplift_range, real_users_data)

for s in sensitivity:
    print(f"Twin uplift: {s['twin_uplift']:.0%} → LTV: {s['projected_ltv']:.0f} ({s['ltv_uplift']:+.1f}%)")
```

## 5.3.4. Валидация модели

```python
def validate_calibration(twin_data, real_data):
    """
    Валидация калибровочного коэффициента
    """
    # Расчёт корреляции между предсказаниями Twin и реальным поведением
    correlation = np.corrcoef(twin_data['predicted'], real_data['actual'])[0, 1]
    
    # Проверка значимости корреляции
    n = len(twin_data['predicted'])
    t_stat = correlation * np.sqrt((n-2) / (1 - correlation**2))
    p_value = 2 * (1 - stats.t.cdf(abs(t_stat), n-2))
    
    return {
        'correlation': correlation,
        't_statistic': t_stat,
        'p_value': p_value,
        'is_valid': p_value < 0.05 and correlation > 0.3
    }
```
